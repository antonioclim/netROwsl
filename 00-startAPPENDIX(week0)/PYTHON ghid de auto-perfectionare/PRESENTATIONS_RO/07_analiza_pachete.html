# Header IP raw (20 bytes minim)

            b'\xb1\xe6\xc0\xa8\x01\x01\xc0\xa8\x01\x02'

# Unpacking (! = network byte order)

print(f"TTL: {ttl}")

print(f"Source: {src_ip}")
print(f"Destination: {dst_ip}")

def parse_tcp_header(data: bytes) -> dict:
    """ParseazƒÉ primii 20 bytes ai header-ului TCP."""
    
    # Format: src_port(H), dst_port(H), seq(I), ack(I), 
    #         offset_flags(H), window(H), checksum(H), urgent(H)
    fields = struct.unpack('!HHIIHHHH', data[:20])
    
    offset_flags = fields[4]
    data_offset = (offset_flags >> 12) * 4  # √Æn bytes
    
    # Extrage flag-urile (ultimii 6 bi»õi)
    flags = {
        'URG': bool(offset_flags & 0x20),
        'ACK': bool(offset_flags & 0x10),
        'PSH': bool(offset_flags & 0x08),
        'RST': bool(offset_flags & 0x04),
        'SYN': bool(offset_flags & 0x02),
        'FIN': bool(offset_flags & 0x01),
    }
    
    return {
        'src_port': fields[0],
        'dst_port': fields[1],
        'seq': fields[2],
        'ack': fields[3],
        'header_len': data_offset,
        'flags': flags,
        'window': fields[5],
    }
