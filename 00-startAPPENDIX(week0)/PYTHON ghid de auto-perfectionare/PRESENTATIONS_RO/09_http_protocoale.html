def parse_http_response(raw: str) -> dict:
    """ParseazƒÉ un rƒÉspuns HTTP raw."""
    
    # SeparƒÉ headers de body
    header_section, body = raw.split("\r\n\r\n", 1)
    lines = header_section.split("\r\n")
    
    # Prima linie: status
    status_line = lines[0]
    parts = status_line.split(" ", 2)
    version, status_code, reason = parts[0], int(parts[1]), parts[2]
    
    # Headers: key: value
    headers = {}
    for line in lines[1:]:
        if ":" in line:
            key, value = line.split(":", 1)
            headers[key.strip().lower()] = value.strip()
    
    return {
        "version": version,
        "status_code": status_code,
        "reason": reason,
        "headers": headers,
        "body": body
    }

# Utilizare
result = parse_http_response(response)
print(f"Status: {result['status_code']}")
print(f"Content-Type: {result['headers'].get('content-type')}")
